71 EulerAngles Quaternion::ToSpherical() const {
72  // Working out these equations would be a pain in the ass. Thankfully, this wikipedia page:
73  // https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles#Quaternion_to_Euler_angles_conversion
74  // uses almost exactly the same euler angle scheme that we do, so we copy their equations almost
75  // wholesale. The only differences are that 1, we use de and roll in the opposite directions,
76  // and 2, we store the conjugate of the quaternion (double check why?), which means we need to
77  // invert the final de and roll terms, as well as negate all the terms involving a mix between
78  // the real and imaginary parts.
79  float ra = atan2(2*(-real*k+i*j), 1-2*(j*j+k*k));
80  if (ra < 0)
81  ra += 2*M_PI;
82  float de = -asin(2*(-real*j-i*k)); // allow de to be positive or negaive, as is convention
83  float roll = -atan2(2*(-real*i+j*k), 1-2*(i*i+j*j));
84  if (roll < 0)
85  roll += 2*M_PI;
86 
87  return EulerAngles(ra, de, roll);
88 }

90 Quaternion SphericalToQuaternion(float ra, float dec, float roll) {
91  assert(roll >= 0.0 && roll <= 2*M_PI);
92  assert(ra >= 0 && ra <= 2*M_PI);
93  assert(dec >= -M_PI && dec <= M_PI);
94 
95  // when we are modifying the coordinate axes, the quaternion multiplication works so that the
96  // rotations are applied from left to right. This is the opposite as for modifying vectors.
97 
98  // It is indeed correct that a positive rotation in our right-handed coordinate frame is in the
99  // clockwise direction when looking down/through the axis of rotation. Just like the right hand
100  // rule for magnetic field around a current-carrying conductor.
101  Quaternion a = Quaternion({ 0, 0, 1 }, ra);
102  Quaternion b = Quaternion({ 0, 1, 0 }, -dec);
103  Quaternion c = Quaternion({ 1, 0, 0 }, -roll);
104  Quaternion result = (a*b*c).Conjugate();
105  assert(result.IsUnit(0.00001));
106  return result;
107 }

283 Quaternion DCMToQuaternion(const Mat3 &dcm) {
284  // Make a quaternion that rotates the reference frame X-axis into the dcm's X-axis, just like
285  // the DCM itself does
286  Vec3 oldXAxis = Vec3({1, 0, 0});
287  Vec3 newXAxis = dcm.Column(0); // this is where oldXAxis is mapped to
288  assert(abs(newXAxis.Magnitude()-1) < 0.001);
289  Vec3 xAlignAxis = oldXAxis.crossProduct(newXAxis).Normalize();
290  float xAlignAngle = AngleUnit(oldXAxis, newXAxis);
291  Quaternion xAlign(xAlignAxis, xAlignAngle);
292 
293  // Make a quaternion that will rotate the Y-axis into place
294  Vec3 oldYAxis = xAlign.Rotate({0, 1, 0});
295  Vec3 newYAxis = dcm.Column(1);
296  // we still need to take the cross product, because acos returns a value in [0,pi], and thus we
297  // need to know which direction to rotate before we rotate. We do this by checking if the cross
298  // product of old and new y axes is in the same direction as the new X axis.
299  bool rotateClockwise = oldYAxis.crossProduct(newYAxis) * newXAxis > 0; // * is dot product
300  Quaternion yAlign({1, 0, 0}, AngleUnit(oldYAxis, newYAxis) * (rotateClockwise ? 1 : -1));
301 
302  // We're done! There's no need to worry about the Z-axis because the handed-ness of the
303  // coordinate system is always preserved, which means the Z-axis is uniquely determined as the
304  // cross product of the X- and Y-axes. This goes to show that DCMs store redundant information.
305 
306  // we want the y alignment to have memory of X, which means we put its multiplication on the
307  // right
308  return xAlign*yAlign;
309 }
