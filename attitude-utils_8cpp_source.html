107 EulerAngles Quaternion::ToSpherical() const {
108  // Working out these equations would be a pain in the ass. Thankfully, this wikipedia page:
109  // https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles#Quaternion_to_Euler_angles_conversion
110  // uses almost exactly the same euler angle scheme that we do, so we copy their equations almost
111  // wholesale. The only differences are that 1, we use de and roll in the opposite directions,
112  // and 2, we store the conjugate of the quaternion (double check why?), which means we need to
113  // invert the final de and roll terms, as well as negate all the terms involving a mix between
114  // the real and imaginary parts.
115  float ra = atan2(2*(-real*k+i*j), 1-2*(j*j+k*k));
116  if (ra < 0)
117  ra += 2*M_PI;
118  float de = -asin(2*(-real*j-i*k)); // allow de to be positive or negaive, as is convention
119  float roll = -atan2(2*(-real*i+j*k), 1-2*(i*i+j*j));
120  if (roll < 0)
121  roll += 2*M_PI;
122 
123  return EulerAngles(ra, de, roll);
124 }

126 Quaternion SphericalToQuaternion(float ra, float dec, float roll) {
127  assert(roll >= 0.0 && roll <= 2*M_PI);
128  assert(ra >= 0 && ra <= 2*M_PI);
129  assert(dec >= -M_PI && dec <= M_PI);
130 
131  // when we are modifying the coordinate axes, the quaternion multiplication works so that the
132  // rotations are applied from left to right. This is the opposite as for modifying vectors.
133 
134  // It is indeed correct that a positive rotation in our right-handed coordinate frame is in the
135  // clockwise direction when looking down/through the axis of rotation. Just like the right hand
136  // rule for magnetic field around a current-carrying conductor.
137  Quaternion a = Quaternion({ 0, 0, 1 }, ra);
138  Quaternion b = Quaternion({ 0, 1, 0 }, -dec);
139  Quaternion c = Quaternion({ 1, 0, 0 }, -roll);
140  Quaternion result = (a*b*c).Conjugate();
141  assert(result.IsUnit(0.00001));
142  return result;
143 }

402 Quaternion DCMToQuaternion(const Mat3 &dcm) {
403  // Make a quaternion that rotates the reference frame X-axis into the dcm's X-axis, just like
404  // the DCM itself does
405  Vec3 oldXAxis = Vec3({1, 0, 0});
406  Vec3 newXAxis = dcm.Column(0); // this is where oldXAxis is mapped to
407  assert(abs(newXAxis.Magnitude()-1) < 0.001);
408  Vec3 xAlignAxis = oldXAxis.crossProduct(newXAxis).Normalize();
409  float xAlignAngle = AngleUnit(oldXAxis, newXAxis);
410  Quaternion xAlign(xAlignAxis, xAlignAngle);
411 
412  // Make a quaternion that will rotate the Y-axis into place
413  Vec3 oldYAxis = xAlign.Rotate({0, 1, 0});
414  Vec3 newYAxis = dcm.Column(1);
415  // we still need to take the cross product, because acos returns a value in [0,pi], and thus we
416  // need to know which direction to rotate before we rotate. We do this by checking if the cross
417  // product of old and new y axes is in the same direction as the new X axis.
418  bool rotateClockwise = oldYAxis.crossProduct(newYAxis) * newXAxis > 0; // * is dot product
419  Quaternion yAlign({1, 0, 0}, AngleUnit(oldYAxis, newYAxis) * (rotateClockwise ? 1 : -1));
420 
421  // We're done! There's no need to worry about the Z-axis because the handed-ness of the
422  // coordinate system is always preserved, which means the Z-axis is uniquely determined as the
423  // cross product of the X- and Y-axes. This goes to show that DCMs store redundant information.
424 
425  // we want the y alignment to have memory of X, which means we put its multiplication on the
426  // right
427  return xAlign*yAlign;
428 }
