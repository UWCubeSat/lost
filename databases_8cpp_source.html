62 void SerializeKVectorIndex(const std::vector<float> &values, float min, float max, long numBins, unsigned char *buffer) {
63  std::vector<int32_t> kVector(numBins+1); // numBins = length, all elements zero
64  float binWidth = (max - min) / numBins;
65 
66  // generate the k-vector part
67  // Idea: When we find the first star that's across any bin boundary, we want to update all the newly sealed bins
68  long lastBin = 0; // first bin the last star belonged to
69  for (int32_t i = 0; i < (int32_t)values.size(); i++) {
70  if (i > 0) {
71  assert(values[i] >= values[i-1]);
72  }
73  assert(values[i] >= min);
74  assert(values[i] <= max);
75  long thisBin = (long)ceil((values[i] - min) / binWidth); // first bin we belong to
76  assert(thisBin >= 0);
77  assert(thisBin <= numBins); // thisBin == numBins is acceptable since kvector length == numBins + 1
78  // if thisBin > lastBin, then no more stars can be added to those bins between thisBin and lastBin, so set them.
79  for (long bin = lastBin; bin < thisBin; bin++) {
80  kVector[bin] = i; // our index is the number of pairs with shorter distance
81  }
82  lastBin = thisBin;
83  }
84  for (long bin = lastBin; bin <= numBins; bin++) {
85  kVector[bin] = values.size();
86  }
87 
88  // verify kvector
89  int32_t lastBinVal = -1;
90  for (const int32_t &bin : kVector) {
91  assert(bin >= lastBinVal);
92  lastBinVal = bin;
93  }
94 
95  unsigned char *bufferStart = buffer;
96  // metadata fields
97  *(int32_t *)buffer = values.size();
98  buffer += sizeof(int32_t);
99  *(float *)buffer = min;
100  buffer += sizeof(float);
101  *(float *)buffer = max;
102  buffer += sizeof(float);
103  *(int32_t *)buffer = numBins;
104  buffer += sizeof(int32_t);
105 
106  // kvector index field
107  // you could probably do this with memcpy instead, but the explicit loop is necessary for endian
108  // concerns? TODO endianness
109  for (const int32_t &bin : kVector) {
110  *(int32_t *)buffer = bin;
111  buffer += sizeof(int32_t);
112  }
113 
114  // verify length
115  assert(buffer - bufferStart == SerializeLengthKVectorIndex(numBins));
116 }

141 long KVectorIndex::QueryLiberal(float minQueryDistance, float maxQueryDistance, long *upperIndex) const {
142  assert(maxQueryDistance > minQueryDistance);
143  if (maxQueryDistance >= max) {
144  maxQueryDistance = max - 0.00001; // TODO: better way to avoid hitting the bottom bin
145  }
146  if (minQueryDistance <= min) {
147  minQueryDistance = min + 0.00001;
148  }
149  if (minQueryDistance > max || maxQueryDistance < min) {
150  *upperIndex = 0;
151  return 0;
152  }

154  long upperBin = BinFor(maxQueryDistance);

157  // bins[lowerBin-1]=number of pairs <= r < query distance, so it is the index of the
158  // first possible item that might be equal to the query distance

164  }
165  // bins[upperBin]=number of pairs <= r >= query distance
