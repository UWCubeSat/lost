399 static float motionBlurredPixelBrightness(const Vec2 &pixel, const GeneratedStar &generatedStar,
400  float t, float stddev) {
401  const Vec2 &p0 = generatedStar.position;
402  const Vec2 &delta = generatedStar.delta;
403  const Vec2 d0 = p0 - pixel;
404  return generatedStar.peakBrightness
405  * stddev*sqrt(M_PI) / (sqrt(2)*delta.Magnitude())
406  * exp(pow(d0.x*delta.x + d0.y*delta.y, 2) / (2*stddev*stddev*delta.MagnitudeSq())
407  - d0.MagnitudeSq() / (2*stddev*stddev))
408  * erf((t*delta.MagnitudeSq() + d0.x*delta.x + d0.y*delta.y) / (stddev*sqrt(2)*delta.Magnitude()));
409 }

532  // need to convert units to photons per time unit per sample
533  float oversamplingBrightnessFactor = oversamplingPerAxis*oversamplingPerAxis;
534

536  // (and fractional amounts are relative to the corner). When we color a pixel, we ideally
537  // would integrate the intensity of the star over that pixel, but we can make do by sampling
538  // the intensity of the star at the /center/ of the pixel, ie, star.x+.5 and star.y+.5

551  float y = yPixel + (ySample+0.5)/oversamplingPerAxis;

554  if (motionBlurEnabled) {
555  curPhotons =
556  (motionBlurredPixelBrightness({x, y}, star, tEnd, starSpreadStdDev)
557  - motionBlurredPixelBrightness({x, y}, star, tStart, starSpreadStdDev))
558  / oversamplingBrightnessFactor;
559  } else {

561  / oversamplingBrightnessFactor;
562  }

567  }
568  }

576  imageData = std::vector<unsigned char>(image.width*image.height);
577  image.image = imageData.data();

583

758 PipelineOutput Pipeline::Go(const PipelineInput &input) {
759  // Start executing the pipeline at the first stage that has both input and an algorithm. From
760  // there, execute each successive stage of the pipeline using the output of the last stage
761  // (human centipede) until there are no more stages set.
762  PipelineOutput result = { 0 };
763 
764  const Image *inputImage = input.InputImage();
765  const Stars *inputStars = input.InputStars();
766  const StarIdentifiers *inputStarIds = input.InputStarIds();
767 
768  // if database is provided, that's where we get catalog from.
769  if (database) {
770  MultiDatabase multiDatabase(database.get());
771  const unsigned char *catalogBuffer = multiDatabase.SubDatabasePointer(kCatalogMagicValue);
772  if (catalogBuffer != NULL) {
773  result.catalog = DeserializeCatalog(multiDatabase.SubDatabasePointer(kCatalogMagicValue), NULL, NULL);
774  } else {
775  std::cerr << "Warning: That database does not include a catalog. Proceeding with the full catalog." << std::endl;
776  result.catalog = input.GetCatalog();
777  }
778  } else {
779  result.catalog = input.GetCatalog();
780  }
781 
782  if (centroidAlgorithm && inputImage) {
783  // TODO: we should probably modify Go to just take an image argument
784  Stars unfilteredStars = centroidAlgorithm->Go(inputImage->image, inputImage->width, inputImage->height);
785  Stars *filteredStars = new std::vector<Star>();
786  for (const Star &star : unfilteredStars) {
787  if (star.magnitude >= centroidMinMagnitude) {
788  filteredStars->push_back(star);
789  }
790  }
791  result.stars = std::unique_ptr<Stars>(filteredStars);
792  inputStars = filteredStars;
793 
794  // any starid set up to this point needs to be discarded, because it's based on input
795  // centroids instead of our new centroids.
796  inputStarIds = NULL;
797  result.starIds = NULL;
798  }
799 
800  if (starIdAlgorithm && database && inputStars && input.InputCamera()) {
801  // TODO: don't copy the vector!
802  result.starIds = std::unique_ptr<StarIdentifiers>(new std::vector<StarIdentifier>(
803  starIdAlgorithm->Go(database.get(), *inputStars, result.catalog, *input.InputCamera())));
804  inputStarIds = result.starIds.get();
805  }
806 
807  if (attitudeEstimationAlgorithm && inputStarIds && input.InputCamera()) {
808  assert(inputStars); // ensure that starIds doesn't exist without stars
809  result.attitude = std::unique_ptr<Attitude>(
810  new Attitude(attitudeEstimationAlgorithm->Go(*input.InputCamera(), *inputStars, result.catalog, *inputStarIds)));
811  }
812 
813  return result;
814 }

859 static std::vector<int> FindClosestCentroids(float threshold,
860  const Stars &one,
861  const Stars &two) {
862  std::vector<int> result;

865  float closestDistance = INFINITY;

870  if (currDistance < threshold && currDistance < closestDistance) {

887 CentroidComparison CentroidsCompare(float threshold,
888  const Stars &expected,
889  const Stars &actual) {
890 
891  CentroidComparison result;
892  // maps from indexes in each list to the closest centroid from other list
893  std::vector<int> expectedToActual, actualToExpected;
894 
895  expectedToActual = FindClosestCentroids(threshold, expected, actual);
896  actualToExpected = FindClosestCentroids(threshold, actual, expected);
897 
898  // any expected stars whose closest actual star does not refer to them are missing
899  for (int i = 0; i < (int)expectedToActual.size(); i++) {
900  if (expectedToActual[i] == -1 ||
901  actualToExpected[expectedToActual[i]] != i) {
902  result.numMissingStars++;
903  } else {
904  result.meanError += Distance(expected[i].position, actual[expectedToActual[i]].position);
905  }
906  }
907  result.meanError /= (expected.size() - result.numMissingStars);
908 
909  // any actual star whose closest expected star does not refer to them is extra
910  for (int i = 0; i < (int)actual.size(); i++) {
911  if (actualToExpected[i] == -1 ||
912  expectedToActual[actualToExpected[i]] != i) {
913  result.numExtraStars++;
914  }
915  }
916 
917  return result;
918 }

946 StarIdComparison StarIdsCompare(const StarIdentifiers &expected, const StarIdentifiers &actual,
947  // use these to map indices to names for the respective lists of StarIdentifiers
948  const Catalog &expectedCatalog, const Catalog &actualCatalog,
949  // stars are ignored if threshold<0
950  float centroidThreshold,
951  const Stars *expectedStars, const Stars *actualStars) {
952 
953  StarIdComparison result = {
954  0, // correct
955  0, // incorrect
956  (int)expected.size() // total
957  };
958  StarIdentifiers expectedSorted = expected;
959  StarIdentifiers actualSorted;
960  if (expectedStars && actualStars) {
961  // map from actual centroid index -> expected centroid index
962  std::vector<int> closestCentroids =
963  FindClosestCentroids(centroidThreshold, *actualStars, *expectedStars);
964  // ensure that even if two actual centroids are close to the same expected centroid, we
965  // don't double count.
966  std::vector<bool> expectedCentroidsUsed(expected.size());
967 
968  for (const StarIdentifier &starId : actual) {
969  int closestIndex = closestCentroids[starId.starIndex];
970 
971  if (closestIndex == -1) {
972  result.numIncorrect++;
973  } else if (!expectedCentroidsUsed[closestIndex]) {
974  expectedCentroidsUsed[closestIndex] = true;
975  actualSorted.push_back(
976  StarIdentifier(closestIndex, starId.catalogIndex, starId.weight));
977  }
978  }
979  } else {
980  actualSorted = actual;
981  }
982 
983  sort(expectedSorted.begin(), expectedSorted.end(), StarIdCompare);
984  sort(actualSorted.begin(), actualSorted.end(), StarIdCompare);
985 
986  auto currActual = actualSorted.cbegin();
987  for (const StarIdentifier &currExpected : expectedSorted) {
988  if (currActual != actualSorted.cend() &&
989  currActual->starIndex == currExpected.starIndex) {
990 
991  if (actualCatalog[currActual->catalogIndex].name == expectedCatalog[currExpected.catalogIndex].name) {
992  result.numCorrect++;
993  } else {
994 #if LOST_DEBUG > 2
995  std::cout << "Expected: " << expectedCatalog[currExpected.catalogIndex].name
996  << " Actual: " << actualCatalog[currActual->catalogIndex].name << std::endl;
997 #endif
998  result.numIncorrect++;
999  }
1000 
1001  currActual++;
1002  }
1003  }
1004 
1005  result.fractionCorrect = (float)result.numCorrect / result.numTotal;
1006  result.fractionIncorrect = (float)result.numIncorrect / result.numTotal;
1007 
1008  return result;
1009 }
