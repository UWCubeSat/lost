570 StarIdentifiers PyramidStarIdAlgorithm::Go(
571  const unsigned char *database, const Stars &stars, const Catalog &catalog, const Camera &camera) const {
572 
573  StarIdentifiers identified;
574  MultiDatabase multiDatabase(database);
575  const unsigned char *databaseBuffer = multiDatabase.SubDatabasePointer(PairDistanceKVectorDatabase::kMagicValue);
576  if (databaseBuffer == NULL || stars.size() < 4) {
577  std::cerr << "Not enough stars, or database missing." << std::endl;
578  return identified;
579  }
580  PairDistanceKVectorDatabase vectorDatabase(databaseBuffer);
581 
582  // smallest normal single-precision float is around 10^-38 so we should be all good. See
583  // Analytic_Star_Pattern_Probability on the HSL wiki for details.
584  float expectedMismatchesConstant = pow(numFalseStars, 4) * pow(tolerance, 5) / 2 / pow(M_PI, 2);
585 
586  // this iteration technique is described in the Pyramid paper. Briefly: i will always be the
587  // lowest index, then dj and dk are how many indexes ahead the j-th star is from the i-th, and
588  // k-th from the j-th. In addition, we here add some other numbers so that the pyramids are not
589  // weird lines in wide FOV images. TODO: Select the starting points to ensure that the first pyramids are all within measurement tolerance.
590  int numStars = (int)stars.size();
591  // the idea is that the square root is about across the FOV horizontally
592  int across = floor(sqrt(numStars))*2;
593  int halfwayAcross = floor(sqrt(numStars)/2);
594  long totalIterations = 0;
595 
596  int jMax = numStars - 3;
597  for (int jIter = 0; jIter < jMax; jIter++) {
598  int dj = 1+(jIter+halfwayAcross)%jMax;
599 
600  int kMax = numStars-dj-2;
601  for (int kIter = 0; kIter < kMax; kIter++) {
602  int dk = 1+(kIter+across)%kMax;
603 
604  int rMax = numStars-dj-dk-1;
605  for (int rIter = 0; rIter < rMax; rIter++) {
606  int dr = 1+(rIter+halfwayAcross)%rMax;
607 
608  int iMax = numStars-dj-dk-dr-1;
609  for (int iIter = 0; iIter <= iMax; iIter++) {
610  int i = (iIter + iMax/2)%(iMax+1); // start near the center of the photo
611 
612  // identification failure due to cutoff
613  if (++totalIterations > cutoff) {
614  std::cerr << "Cutoff reached." << std::endl;
615  return identified;
616  }
617 
618  int j = i+dj;
619  int k = j+dk;
620  int r = k+dr;
621 
622  assert(i != j && j != k && k != r && i != k && i != r && j != r);
623 
624  // TODO: move this out of the loop?
625  Vec3 iSpatial = camera.CameraToSpatial(stars[i].position).Normalize();
626  Vec3 jSpatial = camera.CameraToSpatial(stars[j].position).Normalize();
627  Vec3 kSpatial = camera.CameraToSpatial(stars[k].position).Normalize();
628 
629  float ijDist = AngleUnit(iSpatial, jSpatial);
630 
631  float iSinInner = sin(Angle(jSpatial - iSpatial, kSpatial - iSpatial));
632  float jSinInner = sin(Angle(iSpatial - jSpatial, kSpatial - jSpatial));
633  float kSinInner = sin(Angle(iSpatial - kSpatial, jSpatial - kSpatial));
634 
635  // if we made it this far, all 6 angles are confirmed! Now check
636  // that this match would not often occur due to chance.
637  // See Analytic_Star_Pattern_Probability on the HSL wiki for details
638  float expectedMismatches = expectedMismatchesConstant
639  * sin(ijDist)
640  / kSinInner
641  / std::max(std::max(iSinInner, jSinInner), kSinInner);
642 
643  if (expectedMismatches > maxMismatchProbability) {
644  std::cout << "skip: mismatch prob." << std::endl;
645  continue;
646  }
647 
648  Vec3 rSpatial = camera.CameraToSpatial(stars[r].position).Normalize();
649 
650  // sign of determinant, to detect flipped patterns
651  bool spectralTorch = iSpatial.CrossProduct(jSpatial)*kSpatial > 0;
652 
653  float ikDist = AngleUnit(iSpatial, kSpatial);
654  float irDist = AngleUnit(iSpatial, rSpatial);
655  float jkDist = AngleUnit(jSpatial, kSpatial);
656  float jrDist = AngleUnit(jSpatial, rSpatial);
657  float krDist = AngleUnit(kSpatial, rSpatial); // TODO: we don't really need to
658  // check krDist, if k has been
659  // verified by i and j it's fine.
660 
661  // we check the distances with the extra tolerance requirement to ensure that
662  // there isn't some pyramid that's just outside the database's bounds, but
663  // within measurement tolerance of the observed pyramid, since that would
664  // possibly cause a non-unique pyramid to be identified as unique.
665 #define _CHECK_DISTANCE(_dist) if (_dist < vectorDatabase.MinDistance() + tolerance || _dist > vectorDatabase.MaxDistance() - tolerance) { continue; }
666  _CHECK_DISTANCE(ikDist);
667  _CHECK_DISTANCE(irDist);
668  _CHECK_DISTANCE(jkDist);
669  _CHECK_DISTANCE(jrDist);
670  _CHECK_DISTANCE(krDist);
671 #undef _CHECK_DISTANCE
672 
673  const int16_t *ijEnd, *ikEnd, *irEnd;
674  const int16_t *const ijQuery = vectorDatabase.FindPairsLiberal(ijDist - tolerance, ijDist + tolerance, &ijEnd);
675  const int16_t *const ikQuery = vectorDatabase.FindPairsLiberal(ikDist - tolerance, ikDist + tolerance, &ikEnd);
676  const int16_t *const irQuery = vectorDatabase.FindPairsLiberal(irDist - tolerance, irDist + tolerance, &irEnd);
677 
678  std::unordered_multimap<int16_t, int16_t> ikMap = PairDistanceQueryToMap(ikQuery, ikEnd);
679  std::unordered_multimap<int16_t, int16_t> irMap = PairDistanceQueryToMap(irQuery, irEnd);
680 
681  int iMatch = -1, jMatch = -1, kMatch = -1, rMatch = -1;
682  for (const int16_t *iCandidateQuery = ijQuery; iCandidateQuery != ijEnd; iCandidateQuery++) {
683  int iCandidate = *iCandidateQuery;
684  // depending on parity, the first or second star in the pair is the "other" one
685  int jCandidate = (iCandidateQuery - ijQuery) % 2 == 0
686  ? iCandidateQuery[1]
687  : iCandidateQuery[-1];
688 
689  const Vec3 &iCandidateSpatial = catalog[iCandidate].spatial;
690  const Vec3 &jCandidateSpatial = catalog[jCandidate].spatial;
691 
692  Vec3 ijCandidateCross = iCandidateSpatial.CrossProduct(jCandidateSpatial);
693 
694  for (auto kCandidateIt = ikMap.equal_range(iCandidate); kCandidateIt.first != kCandidateIt.second; kCandidateIt.first++) {
695  // kCandidate.first is iterator, then ->second is the value (other star)
696  int kCandidate = kCandidateIt.first->second;
697  Vec3 kCandidateSpatial = catalog[kCandidate].spatial;
698  bool candidateSpectralTorch = ijCandidateCross*kCandidateSpatial > 0;
699  // checking the spectral-ity early to fail fast
700  if (candidateSpectralTorch != spectralTorch) {
701  continue;
702  }
703 
704  // small optimization: We can calculate jk before iterating through r, so we will!
705  float jkCandidateDist = AngleUnit(jCandidateSpatial, kCandidateSpatial);
706  if (jkCandidateDist < jkDist - tolerance || jkCandidateDist > jkDist + tolerance) {
707  continue;
708  }
709 
710  // TODO: if there are no jr matches, there's no reason to
711  // continue iterating through all the other k-s. Possibly
712  // enumarete all r matches, according to ir, before this loop
713  for (auto rCandidateIt = irMap.equal_range(iCandidate); rCandidateIt.first != rCandidateIt.second; rCandidateIt.first++) {
714  int rCandidate = rCandidateIt.first->second;
715  const Vec3 &rCandidateSpatial = catalog[rCandidate].spatial;
716  float jrCandidateDist = AngleUnit(jCandidateSpatial, rCandidateSpatial);
717  float krCandidateDist;
718  if (jrCandidateDist < jrDist - tolerance || jrCandidateDist > jrDist + tolerance) {
719  continue;
720  }
721  krCandidateDist = AngleUnit(kCandidateSpatial, rCandidateSpatial);
722  if (krCandidateDist < krDist - tolerance || krCandidateDist > krDist + tolerance) {
723  continue;
724  }
725 
726  // we have a match!
727 
728  if (iMatch == -1) {
729  iMatch = iCandidate;
730  jMatch = jCandidate;
731  kMatch = kCandidate;
732  rMatch = rCandidate;
733  } else {
734  // uh-oh, stinky!
735  // TODO: test duplicate detection, it's hard to cause it in the real catalog...
736  std::cerr << "Pyramid not unique, skipping..." << std::endl;
737  goto sensorContinue;
738  }
739  }
740  }
741 
742 
743  }
744 
745  if (iMatch != -1) {
746  printf("Matched unique pyramid!\nExpected mismatches: %e\n", expectedMismatches);
747  identified.push_back(StarIdentifier(i, iMatch));
748  identified.push_back(StarIdentifier(j, jMatch));
749  identified.push_back(StarIdentifier(k, kMatch));
750  identified.push_back(StarIdentifier(r, rMatch));
751 
752  int numAdditionallyIdentified = IdentifyRemainingStarsPairDistance(&identified, stars, vectorDatabase, catalog, camera, tolerance);
753  printf("Identified an additional %d stars.\n", numAdditionallyIdentified);
754  assert(numAdditionallyIdentified == (int)identified.size()-4);
755 
756  return identified;
757  }
758 
759  sensorContinue:;
760  }
761  }
762  }
763  }
764 
765  std::cerr << "Tried all pyramids; none matched." << std::endl;
766  return identified;
767 }
