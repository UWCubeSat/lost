26

28  // Calculate matrix B = sum({ai}{bi}{ri}T)
29  B += ri * bi.transpose() * s.weight;
30  }
31

34  //sigma = B[0][0] + B[1][1] + B[2][2]
35  float sigma = B.trace();
36  //Z = [[B[1][2] - B[2][1]], [B[2][0] - B[0][2]], [B[0][1] - B[1][0]]]
37  Eigen::Vector3f Z;
38  Z << B(1,2) - B(2,1),
39  B(2,0) - B(0,2),
40  B(0,1) - B(1,0);

153 Attitude QuestAlgorithm::Go(const Camera &camera,
154  const Stars &stars,
155  const Catalog &catalog,
156  const StarIdentifiers &starIdentifiers) {
157 
158  // create matrix
159  DqmQuestHelperMatrices m = DqmQuestHelperMatricesConstructor(camera, stars, catalog, starIdentifiers);
160 
161  // calculate coefficients for characteristic polynomial
162  float delta = m.S.determinant();
163  float kappa = (delta * m.S.inverse()).trace();
164  float a = pow(m.sigma,2) - kappa;
165  float b = pow(m.sigma,2) + (m.Z.transpose() * m.Z);
166  float c = delta + (m.Z.transpose() * m.S * m.Z);
167  float d = m.Z.transpose() * (m.S * m.S) * m.Z;
168 
169  // sum up the weights for initial guess of eigenvalue
170  float guess = 0;
171  for (const StarIdentifier &s : starIdentifiers) {
172  guess += s.weight;
173  }
174 
175  // Newton-Raphson method for estimating the largest eigenvalue
176  float eig = QuestEigenvalueEstimator(guess, a, b, c, d, m.sigma);
177 
178  // solve for the optimal quaternion: from https://ahrs.readthedocs.io/en/latest/filters/quest.html
179  float alpha = pow(eig,2) - pow(m.sigma, 2) + kappa;
180  float beta = eig - m.sigma;
181  float gamma = (eig + m.sigma) * alpha - delta;
182 
183  Eigen::Vector3f X = ((alpha * Eigen::Matrix3f::Identity()) + (beta * m.S) + (m.S * m.S)) * m.Z;
184 
185  Eigen::Vector4f optQuat;
186  optQuat << gamma, X[0], X[1], X[2];
187  optQuat /= optQuat.norm();
188 
189  return Attitude(Quaternion(optQuat[0], optQuat[1], optQuat[2], optQuat[3]));
190 }
